\newcommand\newl@ng[2]{%
  \expandafter\gdef\csname l@ng#1\endcsname{%
    #2%
  }%
}%

\newcommand\isolang[0]{%
  \expandafter\ifx\csname l@ng\languagename\endcsname\relax
  \PackageWarning{fkslang}{Undefined language \languagename}%
  null%
  \else
  \expandafter\csname l@ng\languagename\endcsname\fi
}

\newl@ng{czech}{cs}
\newl@ng{slovak}{sk}
\newl@ng{english}{en}
